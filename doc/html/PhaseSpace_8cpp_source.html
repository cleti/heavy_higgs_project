227 // set the 4-vector components for p1,p2 -> k1,k2 process
228 // as seen in the p1-p2 / k1-k2 z.m.f., where p1 defines the z-direction !!!
229 // independent phase space variables are the c.m.e. rs=sqrt((p1+p2)^2)
230 // and the scattering angles y=cos(theta),phi
231 int PS_2_2::set(double const& rs,
232  double const& y,
233  double const& phi)
234 {
235  // check if the setting is physical, i.e. sqrt(s)>m1+m2
236  if ( (rs-sqrt(d_msq[0])-sqrt(d_msq[1]))<EPS_BETA2) return 0;
238  // set up incoming momentum 4-vectors
239  PS_2::set_initial_state(rs);
241 
242  // compute outgoing particle energies and momentum modulus
243  double dm2= d_msq[0]-d_msq[1];
244  double P = 0.5*sqrt(lambda(rs*rs,d_msq[0],d_msq[1]))/rs;
245  double E1 = rs - 0.5*(rs-dm2/rs);
246  double E2 = 0.5*(rs-dm2/rs);
247  double sy = sqrt(1.0-y*y);
248 
249 #ifdef DEBUG
250  if ( std::isnan(P) || std::isinf(P) )
251  {
252  std::cout << std::endl << get_name() << std::endl;
253  WARNING("P is inf/NaN, this should better not happen too often!");
254  PRINT(lambda(rs*rs,d_msq[0],d_msq[1]));
255  SLEEP(3);
256  return 0;
257  }
258  if ( std::isnan(E1) || std::isinf(E1) )
259  {
260  std::cout << std::endl << get_name() << std::endl;
261  WARNING("E1 is inf/NaN, too small value for sqrt(s)?");
262  PRINT(rs);
263  SLEEP(3);
264  return 0;
265  }
266  if ( std::isnan(sy) || std::isinf(sy) )
267  {
268  std::cout << std::endl << get_name() << std::endl;
269  WARNING("sy is inf/NaN, incorrect value for y!");
270  PRINT(1.0-y*y);
271  SLEEP(3);
272  return 0;
273  }
274 #endif
275 
276  if (phi==0.0)
277  {
278  // set up the two outgoing momentum 4-vectors
279  k[0][0] = E1;
280  k[0][1] = P*sy;
281  k[0][2] = 0.0;
282  k[0][3] = P*y;
283 
284  k[1][0] = E2;
285  k[1][1] = -P*sy;
286  k[1][2] = 0.0;
287  k[1][3] = -P*y;
288  }
289  else
290  {
291  // polar angle
292  double cphi = cos(phi);
293  // need to switch sign for phi>pi if we compute sin(phi) in this way
294  double sphi = sqrt(1.0-cphi*cphi);
295  if (phi>Constants::Pi) sphi *= (-1);
296 
297  // set up the two outgoing momentum 4-vectors
298  k[0][0] = E1;
299  k[0][1] = P*sy*cphi;
300  k[0][2] = P*sy*sphi;
301  k[0][3] = P*y;
302 
303  k[1][0] = E2;
304  k[1][1] = -P*sy*cphi;
305  k[1][2] = -P*sy*sphi;
306  k[1][3] = -P*y;
307  }
308 
309  // store phase space variables and compute invariants
310  d_rs = rs;
311  d_beta[0]= P/E1;
312  d_beta[1]= P/E2;
313  d_y = y;
314  d_phi = phi;
315  d_t11 = 2.0*sp(p[0],k[0]);
316  d_t12 = 2.0*sp(p[0],k[1]);
317  d_beta_y = d_y*d_beta[0];
318 
319  // store store phase space density/weight for later use
320  d_wgt = P/(4.0*rs*pow(Constants::TwoPi,2));
321 
322  return 1;
323 }

325 /*
326  this function assumes that the 4-vector components of p1,p2,k1,k2
327  have been set up correctly prior to this call and that they are given in
328  the p1-p2 / k1-k2 z.m.f. , to get this ps object in a consistent state
329  we then have to derive masses, invariants and other phase space variables
330 */
331 int PS_2_2::set(double const& x)
332 {
333  // no checks at all !!!
334  double s = (2.0*sp(p[0],p[1]));
335 
336 #ifdef DEBUG
337  if (s<=0.0)
338  {
339  WARNING("s is <=0!");
340  SLEEP(3);
341  return 0;
342  }
343 #endif
344 
345  d_rs = sqrt(s);
346  d_x = x;
347  double m1sq = sp(k[0],k[0]);
348  double m2sq = sp(k[1],k[1]);
349 
350 #ifdef DEBUG
351  if (s<=4.0*std::max(m1sq,m2sq))
352  {
353  std::cout << std::endl << get_name() << std::endl;
354  WARNING("s is below threshold given by the final state masses!");
355  SLEEP(3);
356  return 0;
357  }
358 #endif
359 
360  // outgoing particle masses
361  d_msq[0] = m1sq;
362  d_msq[1] = m2sq;
363  // use invariant description to define beta
364  // correpsonds to P/E in the c.m.f. of k1 and k2 !!!
365  d_beta[0]= sqrt(1.0-4.0*d_msq[0]/s);
366  d_beta[1]= sqrt(1.0-4.0*d_msq[1]/s);
367  d_t11 = 2.0*sp(p[0],k[0]);
368  d_t12 = 2.0*sp(p[0],k[1]);
369  // sp(k1,p2)-sp(k1,p1) = sp(k2,p1) - sp(k1,p1) = s/2*beta*y
370  d_beta_y = (d_t12-d_t11)/s;
371  d_y = d_beta_y/d_beta[0];
372  // not needed at the moment //
373  d_phi = 0.0;
374  return 1;
375 }

406 void PS_2_2::FillDistributions(
407  DistVec& dist_vec,
408  H_Index id,
409  double const& wgt,
410  double const& mScale) const
411 {
412  // at the moment we are in the parton z.m.f.
413  // for observables like rapidity we need the lab frame 4-vectors
414  // in case 2->2: parton z.m.f. = tt z.m.f.
415  static LT boost_to_lab_frame;
416 
417  // lab frame = z.m.f. of P[0] + P[1]
418  if (!boost_to_lab_frame.set_boost(P[0]+P[1],0))
419  {
420  WARNING("could not set boost to lab frame, skipping distributions");
421  return;
422  }
423 
424  // store the lab frame 4-vectors in a new PS_2_3 instance
425  static PS_2_2 ps_lab("ps 2->2, lab frame, static in PS_2_2::FillDistributions");
426  boost_to_lab_frame.apply_cpy(P[0],ps_lab.P1());
427  boost_to_lab_frame.apply_cpy(P[1],ps_lab.P2());
428  boost_to_lab_frame.apply_cpy(k[0],ps_lab.k1());
429  boost_to_lab_frame.apply_cpy(k[1],ps_lab.k2());
430 #ifdef WITH_T_SPIN
431  boost_to_lab_frame.apply_cpy(S[0],ps_lab.s1());
432  boost_to_lab_frame.apply_cpy(S[1],ps_lab.s2());
433  // do not transform s1_r / s2_r, they contain the restframe spin 4-vectors.
434 #endif
435  // NOTE: if there are observables depending on others than the above transformed 4-vectors,
436  // they must be added here!
437 
438  for (auto dist: dist_vec)
439  {
440  HistArray* hist = dist->GetHistograms();
441  double mass_dim = (hist->GetMassDim())?std::pow(mScale,hist->GetMassDim()):1.0;
442  double obs_val = (*dist)(&ps_lab,this)*mass_dim;
443  double avg_val = dist->Avg(&ps_lab,this);
444  dist->GetHistograms()->FillOne(id,obs_val,avg_val*wgt);
445  }
446 }
