22  HiggsModel* hm = ip->higgs_model;
23 
24  double const& mScale2= hm->Scale2();

27  // hadronic c.m.e.
28  double const& s_hadr = ip->s_hadr;
29  // partonic c.m.e., fixed to MH2
30  double const& s_part = ps->get_msq(0);
31  // PDF convoluation variables
32  double const& x0 = x[0];
33  // the 2->1 phase space is proportional to delta(x0*x1*s_hadr-mH2),
34  // x1-integration is eliminated
35  double const& x1 = s_part/(s_hadr*x0);
36

38  //ps->set();
39  FV const& p1 = ps->p1();
40  FV const& p2 = ps->p2();

42 
43  double res_b = 0.0;
44  double res_b_eff = 0.0;
45  double res_v = 0.0;
46  double res_d = 0.0;
47 
48  if (x1<1.0 && x1>0.0)
49  {
50  // efactor TwoPi/(x0*s_hadr) from phase space:
51  // delta(s_part-mH^2) = 1/(x0*s_had)*delta(x1-mH^2/(x0*s_had))
52  // using s_part = x0*x1*s_had
53  double jac_flux_1 = TwoPi/(2.0*s_part*x0*s_hadr);
54 
55 
56  // the PDF want dimensionful quantities in units of GeV (here, s_hadr is in units of mt!)
57  double f1 = ip->pdf->xfxQ2(21, x0, MUF2*mScale2) / x0;
58  double f2 = ip->pdf->xfxQ2(21, x1, MUF2*mScale2) / x1;
59  // include spin/color average of initial gluons and PFDs, convert to units of picobarn
60  double cf = 0.25*PREF_GG*f1*f2*CONV_GeV2i_pbarn/mScale2;
61 
62  // Born (full ggH vertex and effective ggH vertex)
63  if (flags & BOOST_BINARY(0 000 1))
64  {
65  res_b += Eval_B(p1,p2,*hm,0)*jac_flux_1*cf;
66  res_b_eff += Eval_B(p1,p2,*hm,1)*jac_flux_1*cf;
67  }
68 
69  // virtual corrections
70  if (flags & BOOST_BINARY(0 001 0))
71  {
72  //finite part of the one-loop corrections
73  res_v += Eval_V(p1,p2,*hm)*jac_flux_1*cf;
74  //finite part of the integrated dipoles (delta+endpoint terms)
75  res_d += Eval_ID(p1,p2,*hm,x[1])*jac_flux_1*cf;
76  }
77 
78 
79  // modified phase space for the integrated dipoles
80  // is proportional to delta(x*x0*x1*s_hadr-mH2)
81  // in this case we have 2.0*sp(p1_x,p2_x) = x*s_part = mH^2
82  double const& x1_x = s_part/(s_hadr*x0*x[1]);
83  //check if the modified ocnfiguration for integrated dipoles is physical
84  if ( flags & BOOST_BINARY(0 001 0) && x1_x<1.0)
85  {
86  // flux factor 1/(2*s_part), factor 2 pi from trivial integration over phi
87  double jac_flux_x = TwoPi/(2.0*s_part*x[1]*x0*s_hadr);
88  double f1_x = f1;
89  double f2_x = ip->pdf->xfxQ2(21, x1_x, MUF2*mScale2) / x1_x;
90 
91  // include spin/color average of initial gluons and PFDs, convert to units of picobarn
92  double cf_x = 0.25*PREF_GG*f1_x*f2_x*CONV_GeV2i_pbarn/mScale2;
93  res_d += Eval_ID_X(p1,p2,*hm,x[1])*jac_flux_x*cf_x;
94  }
95  }
96  return (res_b + (ip->K)*(res_v + res_d));

103 double Integrand_2_2_pdf(double* x, size_t dim, void* arg)
104 {
105  using namespace Constants;
106 
107  if (unlikely(dim!=3))
108  {
109  ERROR("wrong integral dimension!");
110  }
111 
112  integrand_par* ip = static_cast<integrand_par*> (arg);
113  PS_2_2* ps = dynamic_cast<PS_2_2*> (ip->ps);
114  const ulong& flags = ip->eval_flags;
115  HiggsModel* hm = ip->higgs_model;
116 
117  double const& mScale2= hm->Scale2();
118  double const& MUF2 = hm->MUF2();
119 
120  // hadronic c.m.e.
121  double const& s_hadr = ip->s_hadr;
122  // partonic c.m.e.
123  double s_part = s_hadr*x[0]*x[1];
124 
125  FV const& p1 = ps->p1();
126  FV const& p2 = ps->p2();
127  FV const& p3 = ps->k2();// this should be the final state gluon -> m=0
128 
129  if (ps->set(sqrt(s_part),x[2],0.0))
130  {
131  // cuts on collinear and soft phase space regions (x[2] is the scattering angle)
132  if (fabs(x[2]) > Cuts::COLL_CUT) return 0.0;
133  if (p3[0] < Cuts::SOFT_CUT) return 0.0;
134 
135  double res_r = 0.0;
136  double res_d = 0.0;
137  // flux factor 1/(2*s_part), factor 2 pi from trivial integration over phi
138  double jac_flux = TwoPi*ps->get_wgt()/(2.0*s_part);
139 
140  // the PDF wants dimensionful quantities in units of GeV (here, s_hadr is in units of mt!)
141  double f1_g = ip->pdf->xfxQ2(21, x[0], MUF2*mScale2) / x[0];
142  double f2_g = ip->pdf->xfxQ2(21, x[1], MUF2*mScale2) / x[1];
143 
144  // real corrections gg - unitegrated dipoles
145  if (flags & BOOST_BINARY(0 010 0))
146  {
147  res_r += 0.25*PREF_GG*f1_g*f2_g*jac_flux*(Eval_R_gg(p1,p2,p3,*hm)-Eval_UID(p1,p2,p3,*hm));
148 #ifdef DEBUG
149  CHECKNA(res_r);
150 #endif
151  }
152  // qq and qg contributions
153  if (flags & BOOST_BINARY(1 100 0))
154  {
155  static auto quark_pids = { 5, 4, 3, 2, 1};
156  double qq_pdf = 0.0;
157  double qg_pdf = 0.0;
158  for (auto i: quark_pids)
159  {
160  double f1_q = ip->pdf->xfxQ2(+i, x[0], MUF2*mScale2) / x[0];
161  double f2_qb = ip->pdf->xfxQ2(-i, x[1], MUF2*mScale2) / x[1];
162 
163  // taken into account by a factor of 2 in Eval_sigma_R_qq_fin(..)|
164  qq_pdf += f1_q*f2_qb;//+f1_qb*f2_q; <---------------------------|
165 
166  // taken into account by a factor of 2 in Eval_sigma_R_qg_fin(..)|
167  qg_pdf += f1_q*f2_g+f1_g*f2_qb;// + f2_q*f2_g+f2_g*f1_qb <------|
168  }
169 
170  // MH2 / s
171  double z = ps->get_msq(0) / ps->get_s();
172  if (flags & BOOST_BINARY(0 100 0))
173  {
174  // q q -> H g
175  res_r += qq_pdf*Eval_sigma_R_qq_fin(z,*hm);
176 #ifdef DEBUG
177  CHECKNA(res_r);
178 #endif
179  }
180  if (flags & BOOST_BINARY(1 000 0))
181  {
182  // q g -> H q + qbar g -> H qbar
183  res_r += qg_pdf*Eval_sigma_R_qg_fin(z,*hm);
184 #ifdef DEBUG
185  CHECKNA(res_r);
186 #endif
187  }
188  }
189  // this will only be rescaled if the LO matrix elements were evaluated before !
190  // otherwise K = 1
191  return (ip->K)*(res_r-res_d)*CONV_GeV2i_pbarn/mScale2;
192  }
193  else
194  {
195  return 0.0;
196  }
197 }
