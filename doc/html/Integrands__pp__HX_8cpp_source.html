114 double Integrand_2_2_pdf(double* x, size_t dim, void* arg)
115 {

118 
119  if (unlikely(dim!=3))

124 
125  integrand_par* ip = static_cast<integrand_par*> (arg);
126  PS_2_2* ps = dynamic_cast<PS_2_2*> (ip->ps);
127  unsigned long& flags = ip->eval_flags;
128 
129 
130  // hadronic c.m.e.
131  double s_hadr = ip->s_hadr;
132  // partonic c.m.e.
133  double s_part = s_hadr*x[0]*x[1];
134 
135  FV const& p1 = ps->p1();
136  FV const& p2 = ps->p2();
137  FV const& p3 = ps->k2();// this should be the final state gluon -> m=0
138 
139  if (ps->set(sqrt(s_part),x[2],0.0))
140  {
141  // cuts on collinear and soft phase space regions (x[2] is the scattering angle)
142  if (fabs(x[2]) > Cuts::COLL_CUT) return 0.0;
143  if (p3[0] < Cuts::SOFT_CUT) return 0.0;
144 
145  double res_r = 0.0;
146  double res_d = 0.0;
147  // flux factor 1/(2*s_part), factor 2 pi from trivial integration over phi
148  double jac_flux = TwoPi*ps->get_wgt()/(2.0*s_part);
149 
150  // the PDF wants dimensionful quantities in units of GeV (here, s_hadr is in units of mt!)
151  double f1_g = ip->pdf->xfxQ2(21, x[0], MUF2*mScale2) / x[0];
152  double f2_g = ip->pdf->xfxQ2(21, x[1], MUF2*mScale2) / x[1];
153 
154  // real corrections gg - unitegrated dipoles
155  if (flags & BOOST_BINARY(0 010 0))
156  {
157  res_r += PREF_GG*f1_g*f2_g*jac_flux*(Eval_R_gg(p1,p2,p3)-Eval_UID(p1,p2,p3));
158 #ifdef DEBUG
159  CHECKNA(res_r);
160 #endif
161  }
162  // qq and qg contributions
163  if (flags & BOOST_BINARY(1 100 0))
164  {
165  static auto quark_pids = { 5, 4, 3, 2, 1};
166  double qq_pdf = 0.0;
167  double qg_pdf = 0.0;
168  for (auto i: quark_pids)
169  {
170  double f1_q = ip->pdf->xfxQ2(+i, x[0], MUF2*mScale2) / x[0];
171  double f2_qb = ip->pdf->xfxQ2(-i, x[1], MUF2*mScale2) / x[1];
172  // double f1_qb = ip->pdf->xfxQ2(-i, x[0], MUF2*mScale2) / x[0];
173  // double f2_q = ip->pdf->xfxQ2(+i, x[1], MUF2*mScale2) / x[1];
174  qq_pdf += f1_q*f2_qb;//+f1_qb*f2_q;
175  qg_pdf += f1_q*f2_g+f1_g*f2_qb;
176  }
177 
178  double z = ps->get_msq(0) / (2.0*sp(p1,p2));
179  if (flags & BOOST_BINARY(0 100 0))
180  {
181  //res_r += qq_pdf*PREF_QQ*Eval_R_qq(p1,p2,p3)*jac_flux;
182  res_r += qq_pdf*Eval_sigma_R_qq_fin(z);
183 #ifdef DEBUG
184  CHECKNA(res_r);
185 #endif
186  }
187  if (flags & BOOST_BINARY(1 000 0))
188  {
189  res_r += qg_pdf*Eval_sigma_R_qg_fin(z);
190 #ifdef DEBUG
191  CHECKNA(res_r);
192 #endif
193  }
194  }
195  // this will only be rescaled if the LO matrix elements were evaluated before !
196  // otherwise K = 1
197  return (ip->K)*(res_r-res_d)*CONV_mt2i_pbarn;
198  }
199  else
200  {
201  return 0.0;
202  }
203 }
