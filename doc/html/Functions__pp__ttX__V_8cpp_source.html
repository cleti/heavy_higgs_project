14 // delta Z_M, mass renormalization factor for a quark (MS-bar, 1-loop, 5-flavour)
15 static double ZM (
16  double const& MT2,
17  double const& AlphaS,
18  double const& MUR2)
19 {
20  using namespace Constants;
21  static const int I = 0;
22  static const double ZERO = 0.0;
23  static double res = 0.0;
24  static double MT2_t = 0.0;
25  static double LNMUR2 = 0.0;
26 
27  if (MT2_t != MT2)
28  {
29  LNMUR2 = (RE(qli2_(&MT2,&ZERO,&MT2,&MUR2,&I))-2.0);// (RE(I2_MT2_0_MT2_MU2_0)-2.0) = log(MUR2/MT2)
30  MT2_t = MT2;
31  res = (AlphaS/FourPi*CF)*(-3.0*LNMUR2-4.0);
32  }
33  return res;
34 }

207 double Eval_V(
208  const PS_2_2& ps,
209  HiggsModel& hm,
210  const ulong& flags)
211 {
212  PREFACTORS(hm); // defines ap and hp
213 
214  // these have to be adjusted whenever S changes!!!
215  hm.SetHiggsPrefactors(ps.get_s(),1);
216  // set scalar integrals (update invariants sometime before this call by Set_Invariants())
217  Set_SI(ps.get_msq(0), // assuming that get_msq(1) is the same value
218  ps.get_rs(),
219  ps.get_t11(),
220  ps.get_t12(),
221  hm.MUR2(),
222  flags);
223 
224  double res = 0.0;
225  // add contributions of individual diagrams according to flags in flags
226  if (flags & F_EVAL_V_PHI0xQCD1)
227  {
228  // 2RE[PHI0 x QCD1] interference terms classified by QCD1 diagram type
229  // self-energy diagram including ren. counter term
230  res += Eval_V_SE (ps,ap,hp);
231 #ifdef DEBUG
232  CHECKNAN(res);
233 #endif
234  // 4g vertex diagram
235  res += Eval_V_4G (ps,ap,hp);
236 #ifdef DEBUG
237  CHECKNAN(res);
238 #endif
239  // qg veretx correction (upper and lower vertex) including ren. counter term
240  res += 2.0*Eval_V_V1 (ps,ap,hp);
241 #ifdef DEBUG
242  CHECKNAN(res);
243 #endif
244  // box 1
245  res += Eval_V_B1 (ps,ap,hp);
246 #ifdef DEBUG
247  CHECKNAN(res);
248 #endif
249  // box 2
250  res += Eval_V_B2 (ps,ap,hp);
251 #ifdef DEBUG
252  CHECKNAN(res);
253 #endif
254  // box 3
255  res += Eval_V_B3 (ps,ap,hp);
256 #ifdef DEBUG
257  CHECKNAN(res);
258 #endif
259  }
260  // 2RE[PHI1 x QCD0] interference terms
261  if (flags & F_EVAL_V_PHI1xQCD0)
262  {
263  res += Eval_V_2RE_PHI1xQCD0(ps,ap,hp);
264 #ifdef DEBUG
265  CHECKNAN(res);
266 #endif
267  }
268  // 2RE[PHI1xPHI0] interference terms
269  if (flags & F_EVAL_V_PHIxPHI)
270  {
271  res += Eval_V_PHIxPHI(ps,ap,hp);
272 #ifdef WITH_T_SPIN
273  if (hm.NBosons()>1)
274  {
275  res += Eval_V_PHIxPHI_IM_INTab(ps,ap,hp);
276  }
277 #endif
278 #ifdef DEBUG
279  CHECKNAN(res);
280 #endif
281  }
282  return res;
283 }
