41 TH1D* makeErrorHist(TH1D* hist_l, TH1D* hist_h)
42 {
43  if (!hist_l || !hist_h)
44  {
45  cout << endl << " In " << __FUNCTION__ << ": nullpointer received." << endl;
46  exit(0);
47  }
48  if (hist_l->GetNbinsX() != hist_h->GetNbinsX())
49  {
50  cout << endl << " In " << __FUNCTION__ << ": binning of input histograms does not match." << endl;
51  exit(0);
52  }
53 
54  TH1D* ret = new TH1D(*hist_l);
55  for (int i=1;i<=hist_l->GetNbinsX();++i)
56  {
57  double cnt = (hist_l->GetBinContent(i)+hist_h->GetBinContent(i))/2.0;
58  double err = 2.0*fabs(hist_h->GetBinContent(i)-cnt);
59  ret->SetBinContent(i,cnt);
60  ret->SetBinError(i,err);
61  }
62 
63  ret->SetFillStyle(3002);
64  ret->SetFillColor(1);
65  ret->SetLineWidth(0);
66  return ret;
67 }
