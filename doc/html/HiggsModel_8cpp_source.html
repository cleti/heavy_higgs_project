47 void HiggsBoson::SetFormFactors(
48  double const& S,
49  double const& mt2,
50  double const& mb2)
51 {
52  static int I = 0;// finite part (the integrals here are not divergent)
53  static double zero = 0.0;
54  static double MUR2 = 0.0; // finite integrals do not depend on MUR
55 
56  if (S!=d_S_FF)
57  {
58  d_S_FF = S;
59  // reset values
60  d_FH_full = 0.0;
61  d_FA_full = 0.0;
62 
63  // top contribution
64  if (d_At != 0.0 || d_Bt != 0.0)
65  {
66  c_double I3t = qli3_(&S,&zero,&zero,&mt2,&mt2,&mt2,&MUR2,&I);
67  d_FH_full += mt2 * d_At * ( (S-4.0*mt2) * I3t - 2.0 );
68  d_FA_full += mt2 * d_Bt * S * I3t;
69  }
70  if (mb2>0.0)
71  {
72  // bottom contribution
73  if (d_Ab != 0.0 || d_Bb != 0.0)
74  {
75  c_double I3b = qli3_(&S,&zero,&zero,&mb2,&mb2,&mb2,&MUR2,&I);
76  d_FH_full += mb2 * d_Ab * ( (S-4.0*mb2) * I3b - 2.0 );
77  d_FA_full += mb2 * d_Bb * S * I3b;
78  }
79  }
80  d_FH_full /= -(4.0*S);
81  d_FA_full /= (8.0*S);
82 
83 #ifdef DEBUG
84  if (S<=0.0)
85  {
86  ERROR("S<=0!");
87  }
88  if (std::isnan(RE(d_FH_full)) || std::isinf(RE(d_FH_full)) || std::isnan(IM(d_FH_full)) || std::isinf(IM(d_FH_full)))
89  {
90  PRINT(qli3_(&S,&zero,&zero,&mt2,&mt2,&mt2,&MUR2,&I));
91  PRINT(qli3_(&S,&zero,&zero,&mb2,&mb2,&mb2,&MUR2,&I));
92  ERROR("d_FH_full is inf/nan!");
93  }
94  if (std::isnan(RE(d_FA_full)) || std::isinf(RE(d_FA_full)) || std::isnan(IM(d_FA_full)) || std::isinf(IM(d_FA_full)))
95  {
96  PRINT(qli3_(&S,&zero,&zero,&mt2,&mt2,&mt2,&MUR2,&I));
97  PRINT(qli3_(&S,&zero,&zero,&mb2,&mb2,&mb2,&MUR2,&I));
98  ERROR("d_FA_full is inf/nan!");
99  }
100 #endif
101  }

230 void HiggsModel::SetHiggsPrefactors(double const& S, bool EFF)
231 {
232  // reset
233  d_HPref.Reset();
234 
235  // iterate over d_Bosons and add the individual contributions
236  auto last = std::end(d_Bosons);
237  for (auto phi_i = std::begin(d_Bosons); phi_i!=last; ++phi_i)
238  {
239  if (!EFF) (*phi_i)->SetFormFactors(S,d_mt2,d_mb2);
240  (*phi_i)->SetPropagator(S);
241  c_double const& FHi = (*phi_i)->GetFH(EFF);
242  c_double const& FAi = (*phi_i)->GetFA(EFF);
243  c_double const& Di = (*phi_i)->GetPropagator();
244  double FH2i = std::norm(FHi);
245  double FA2i = std::norm(FAi);
246  double const& di = (*phi_i)->GetPropagatorSq();
247  double const& Ati = (*phi_i)->At();
248  double const& Bti = (*phi_i)->Bt();
249  double At2i = std::pow(Ati,2);
250  double Bt2i = std::pow(Bti,2);
251 
252 
253  // prefactors in PHIxQCD amplitudes
254  // these are just the sum of individual phi contributions
255  d_HPref.At_fH_re += Ati*(FHi*Di).real();
256  d_HPref.At_fA_re += Ati*(FAi*Di).real();
257  d_HPref.Bt_fH_re += Bti*(FHi*Di).real();
258  d_HPref.Bt_fA_re += Bti*(FAi*Di).real();
259 
260  d_HPref.At_fH_im += Ati*(FHi*Di).imag();
261  d_HPref.At_fA_im += Ati*(FAi*Di).imag();
262  d_HPref.Bt_fH_im += Bti*(FHi*Di).imag();
263  d_HPref.Bt_fA_im += Bti*(FAi*Di).imag();
264 
265  // prefactors in PHIxPHI amplitudes
266  // first the sum of the individual phi_i^2 contributions
267  d_HPref.At2_fH2_De += At2i*FH2i*di;
268  d_HPref.At2_fA2_De += At2i*FA2i*di;
269  d_HPref.Bt2_fH2_De += Bt2i*FH2i*di;
270  d_HPref.Bt2_fA2_De += Bt2i*FA2i*di;
271 
272 #ifdef WITH_T_SPIN
273  // only needed for polarized amplitudes:
274  // PHIxPHI terms: add at*bt interference from Phi1^2 contribution
275  d_HPref.At_Bt_fH2_De += Ati*Bti*FH2i*di;
276  d_HPref.At_Bt_fA2_De += Ati*Bti*FA2i*di;
277 #endif
278 
279  // second loop to get the phi^2 contributions
280  for (auto phi_j = phi_i+1; phi_j!=last; ++phi_j)
281  {
282  if (!EFF) (*phi_j)->SetFormFactors(S,d_mt2,d_mb2);
283  (*phi_j)->SetPropagator(S);
284 
285  c_double const& FHj = (*phi_j)->GetFH(EFF);
286  c_double const& FAj = (*phi_j)->GetFA(EFF);
287  c_double const& Dj = (*phi_j)->GetPropagator();
288  double const& Atj = (*phi_j)->At();
289  double const& Btj = (*phi_j)->Bt();
290 
291  c_double DiDj = Di*std::conj(Dj);
292 
293  // prefactors in PHIxPHI amplitudes
294  // here we also have to sum all the phi_i * phi_j interferences
295  d_HPref.At2_fH2_De += 2.0*Ati*Atj*(FHi*std::conj(FHj)*DiDj).real();
296  d_HPref.At2_fA2_De += 2.0*Ati*Atj*(FAi*std::conj(FAj)*DiDj).real();
297  d_HPref.Bt2_fH2_De += 2.0*Bti*Btj*(FHi*std::conj(FHj)*DiDj).real();
298  d_HPref.Bt2_fA2_De += 2.0*Bti*Btj*(FAi*std::conj(FAj)*DiDj).real();
299 #ifdef WITH_T_SPIN
300  // only needed for polarized amplitudes:
301  // PHIxPHI terms: at Phi1 * bt Phi2 interferences
302  d_HPref.At_Bt_fH2_De += (Ati*Btj+Atj*Bti)*(FHi*std::conj(FHj)*DiDj).real();
303  d_HPref.At_Bt_fA2_De += (Ati*Btj+Atj*Bti)*(FAi*std::conj(FAj)*DiDj).real();
304  // the Phi1 At * Phi2 Bt interference is prop. to the imaginary parts
305  d_HPref.At_Bt_fH2_DeIM += (Ati*Btj-Atj*Bti)*(FHi*std::conj(FHj)*DiDj).imag();
306  d_HPref.At_Bt_fA2_DeIM += (Ati*Btj-Atj*Bti)*(FAi*std::conj(FAj)*DiDj).imag();
307 #endif
308  }
309  }
