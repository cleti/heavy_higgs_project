163 double obs_PHI(FV const& k1, FV const& k2)

173 // 2-dim transversal opening angle of the two vectors

177  double K2 = sqrt(k2[1]*k2[1]+k2[2]*k2[2]);

180 // spatial triple product of three 4-vectors k1,k2,k3
181 double obs_TriProd(FV const& k1, FV const& k2, FV const& k3)
182 {
183  // = (k1 x k2) dot k3 / |k1 x k2| / |k3|
184  double t3 = k1[0] * k2[1] - k1[1] * k2[0];
185  double t7 = -k1[0] * k2[2] + k1[2] * k2[0];
186  double t11 = k1[1] * k2[2] - k1[2] * k2[1];
187  double t14 = t3 * t3;
188  double t15 = t7 * t7;
189  double t16 = t11 * t11;
190  double t18 = std::pow(k3[0], 2);
191  double t19 = std::pow(k3[1], 2);
192  double t20 = std::pow(k3[2], 2);
193  double t23 = std::sqrt((t14 + t15 + t16) * (t18 + t19 + t20));
194  return (t11 * k3[0] + t3 * k3[2] + t7 * k3[1]) / t23;
195 }
