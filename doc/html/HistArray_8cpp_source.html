300  // if there is no space or something after the last number

387  bool discardCurrentTable)
388 {
389  if (!file.is_open())
390  {
391  return 0;
392  }
393  // const int colWidth1 = 16;
394  // const int colWidth2 = 20;
395  // alias for the current histgram in 'DistVec* dist_vec'
396  TH1D& hist0 = d_histograms[H_LO_QCD];
397  std::string buff;
398  //std::stringstream sbuff;
399  int rowCount = 0;
400  int rowMax = hist0.GetNbinsX();
401  double binLowEdge;
402  double binVal;
403 
404  // MAIN LOOP ////////////////////////////////////////////////////////////////////////////////
405  while(file.good())
406  {
407  // FETCH INPUT LINE /////////////////////////////////////////////////////////////////////
408  std::getline(file,buff);
409  // if there is no space or something after the last number
410  // iss.good() will be false already after extraction of the last number
411  buff+=' ';
412  // remove leading whitespace
413  while (buff.front() == ' ' || buff.front() == '\t')
414  {
415  buff.erase(std::begin(buff));
416  }
417  // INSPECT LINE ////////////////////////////////////////////////////////////////////////
418  if (buff.size()>0)
419  {
420  if (buff.front() == '#')
421  { // comment-line: comment in combination with rowCount>0 signals end-of-table
422  if (rowCount>0)
423  {
424  std::cout << std::endl;
425  std::cout << " [ END OF TABLE ";
426  std::cout << ", " << rowCount << " rows ]\n\n";
427  return 1;
428  }
429  if (discardCurrentTable) discardCurrentTable = false;
430  }
431  else
432  { // data-line: extract numbers, count table row and compare with target histogram size
433  // discardCurrentTable: do nothing until a comment-line was processed
434  if (discardCurrentTable) continue;
435  // use stringstream to extract numbers
436  std::stringstream sbuff(buff);
437  // first column is the lower bin boundary
438  sbuff >> binLowEdge;
440  if ( ++rowCount > rowMax+1)
441  {
442  std::cout << "\n Error reading row " << rowCount;
443  std::cout << "\n Buffer content (str): " << buff;
444  std::cout << "\n Too many rows in input table!\n\n ";
445  return 0;
446  }
447  // compare bin boundary given in current row with target histogram binning
448  if ( std::fabs( binLowEdge - (hist0.GetBinLowEdge(rowCount)) ) > 1e-5 )
449  {
450  std::cout << "\n Error reading row " << rowCount;
451  std::cout << "\n Buffer content (str): " << buff;
452  std::cout << "\n low-edge hist: " << hist0.GetBinLowEdge(rowCount);
453  std::cout << "\n low-edge input:" << binLowEdge;
454  std::cout << "\n Input data does not match histogram binning!\n\n";
455  // discard data from all histograms
456  //this->Reset();
457  return 0;
458  }
460  // std::cout << std::setw(5) << rowCount << " ";
461  // std::cout << std::setw(colWidth1) << binLowEdge << " ";
462  int colCount = 0;
463  // extract the subsequent columns
464  while (1)
465  {
466  sbuff >> binVal;
467  if (sbuff.good())
468  {
469  // std::cout << std::setw(colWidth2) << binVal << " ";
470  // get histogram index corresponding to colCount
471  H_IndexMap::const_iterator it = imap.find(colCount++);
472  // fill value into histogram if the index was found in the map
473  if (it != imap.end())

475  // std::cout << std::setw(colWidth2/2) << it->second << " ";

481  }
482  }
483  else
484  {
485  break;
486  }
487  }
488  // std::cout << std::endl;
489  }
490  }
491  // !INSPECT LINE ////////////////////////////////////////////////////////////////////////
492  }
493  // !MAIN LOOP ///////////////////////////////////////////////////////////////////////////////
494  if (file.eof())
495  {
496  if (rowCount>0)

501 
502  }
