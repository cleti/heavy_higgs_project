107 // mass of 2-particle system [GeV]
108 double obs_M12(FV const& k1, FV const& k2)
109 {
110  return sqrt(sp(k1,k1)+sp(k2,k2)+2.0*sp(k1,k2))*RunParameters::mScale;
111 }
112 // transverse momentum (transverse plane is the x-y-plane) [GeV]
113 double obs_PT(FV const& k)
114 {
115  return sqrt(k[1]*k[1]+k[2]*k[2])*RunParameters::mScale;
116 }
117 // pseudo-rapidity
118 double obs_Y(FV const& k)
119 {
120  // the rapidity is not invariant under boosts in z-direction !
121  // note that e.g. the reduced dipole phase spaces are boosted along z
122  // with respect to the parton c.m.f.
123  // double K = LEN(k);
124  // return 0.5*log(((K+k[3])/(K-k[3])));
125  return 0.5*log(((k[0]+k[3])/(k[0]-k[3])));
126 }
127 // difference of abs. values of transverse rapidities
128 double obs_DY(FV const& k1, FV const& k2)
129 {
130  return fabs(obs_Y(k1))-fabs(obs_Y(k2));
131 }
132 // 3-dim opening angle of the two vectors
133 double obs_PHI(FV const& k1, FV const& k2)
134 {
135  double K1 = LEN(k1);
136  double K2 = LEN(k2);
137  return (k1[1]*k2[1]+k1[2]*k2[2]+k1[3]*k2[3])/(K1*K2);
138 }
