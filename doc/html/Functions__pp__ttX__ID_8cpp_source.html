41 // evaluate the finite part of the integrated dipoles: x-independent part of I,P,K operators
42 double Eval_ID_GG(
43  const PS_2_2& ps,
44  const double& x,
45  HiggsModel& hm,
46  const ulong& flags)
47 {
48  if (flags & (F_EVAL_D_GG_DELTA | F_EVAL_D_GG_END))
49  {
50  PREFACTORS(hm); // defines ap and hp
51  double res = 0.0;
52 
53  //using namespace RunParameters;
54 
55  const double& s12_1 = ps.get_s();
56 
57  // these have to be adjusted before calls to Eval_??? whenever S changes!!!
58  hm.SetHiggsPrefactors(s12_1,1);
59  // collect the Born contributions
60  // II/FF colour correlations only amount to factors of CF, CA included in Eval_DIP_GG_CONT()
61  // PHI^2
62  double B_phi=0.0;
63  if (flags & F_EVAL_B_PHIxPHI)
64  {
65  B_phi += Eval_B_PHIxPHI(ps,ap,hp);
66  }
67  // PHIxQCD interference
68  double B_int=0.0,BC_int=0.0;
69  if (flags & F_EVAL_B_PHIxQCD)
70  {
71  B_int += Eval_B_2PHIxQCD(ps,ap,hp);
72  // IF and FI colour correlated Born MEs, only PHIxQCD contributes
73  BC_int -= 0.5*ps.get_beta_y()*B_int;
74  }
75 
76  const double& MUR2 = hm.MUR2();
77  const double& MUF2 = hm.MUF2();
78  const double& AlphaS = hm.AlphaS();
79  const double& beta_1 = ps.get_beta();
80  const double S12_1 = 2.0*sp(ps.k1(),ps.k2());
81  const double& t11_1 = ps.get_t11();
82  const double& t12_1 = ps.get_t12();
83 
84  // dipole contributions proportional to delta(1-x) originating from I, P and K operators
85  if (flags & F_EVAL_D_GG_DELTA)
86  {
87  res += Eval_DIP_GG_DELTA_I (s12_1,
88  S12_1,
89  beta_1,
90  t11_1,
91  t12_1,
92  B_phi+B_int,
93  BC_int,
94  MUR2,
95  AlphaS);
96  res += Eval_DIP_GG_DELTA_PK(s12_1,
97  t11_1,
98  t12_1,
99  B_phi+B_int,
100  2.0*B_phi+B_int,
101  BC_int,
102  MUF2,MUR2,
103  AlphaS);
104  }
105  // distribution end-point part (check tech. cut also here)
106  if (flags & F_EVAL_D_GG_END && x<Cuts::IDIP_X_CUT)
107  {
108  res += Eval_DIP_GG_DIST_E(s12_1,
109  t11_1,
110  t12_1,
111  B_phi+B_int, // II and FF colour correlated Born MEs
112  // these are proportional to the uncorrelated Born MEs (factors -CA/CF)
113  2.0*B_phi+B_int,// this is the Born contribution that is multiplied with the K-bar terms
114  // the IF colour correlated Born contribution for PHIxPHI is zero, but the K-bar terms
115  // of the IF dipoles do not involve colour operators thus PHIxPHI contributes!
116  // further the factor 1/2 has to be compensated for the PHIxPHI part
117  // this factor was introduced because half of the IF phiXqcd dipoles were neglected
118  BC_int, // IF and FI colour correlated Born MEs
119  // only the PHIxQCD interference terms contribute here
120  x,
121  MUF2,
122  AlphaS);
123  }
124  return res;
125  }
126  else
127  {
128  return 0.0;
129  }
130 }

136 int Eval_ID_X(
137  const PS_2_2& ps_x,
138  const double& x,
139  HiggsModel& hm,

144  // check technical cut
145  if (x>Cuts::IDIP_X_CUT) return 0;
146 
147  // c.m.e. on boosted phase space
148  double const& s12_x = ps_x.get_s();
149 
150  PREFACTORS(hm); // defines ap and hp
151  // these have to be adjusted before calls to Eval_??? whenever S changes!!!
152  hm.SetHiggsPrefactors(s12_x,1);
153 
154  // collect the Born contributions evaluated at boosted phase space point
155  // II and FF colour correlations only amount to factors of CF, CA included in Eval_DIP_GG_CONT()
156  // PHI^2
157  double B_phi_x = 0.0;
158  if (flags & F_EVAL_B_PHIxPHI)
159  {
160  B_phi_x += Eval_B_PHIxPHI(ps_x,ap,hp);
161  }
162 #ifdef DEBUG
163  CHECKNA(B_phi_x);
164 #endif
165 
166  double B_int_x = 0.0;
167  double BC_int_x= 0.0;
168  // PHIxQCD interference
169  if (flags & F_EVAL_B_PHIxQCD)
170  {
171  B_int_x += Eval_B_2PHIxQCD(ps_x,ap,hp);
172  // IF and FI colour correlated Born MEs, only PHIxQCD contributes
173  BC_int_x+= -0.5*ps_x.get_beta_y()*B_int_x;
174  }
175 #ifdef DEBUG
176  CHECKNA(B_int_x);
177 #endif
178 
179  double const& MUF2 = hm.MUF2();
180  double const& AlphaS = hm.AlphaS();
181  // original partonic c.m.e.
182  double s12_1 = s12_x / x;
183  // invariants t_aj = 2 p_a.k_j, where p_a is the original initial state
184  // momentum and k_j comes from the boosted phase space ps_x
185  // c.f. text below (6.38) in hep-ph/0201036v1
186  double t11_x = ps_x.get_t11() / x;
187  double t12_x = ps_x.get_t12() / x;
188 
189  if ( flags & (F_EVAL_D_GG_CONT) )
190  {
191  res_gg = Eval_DIP_GG_CONT(s12_1,
192  t11_x,
193  t12_x,
194  B_phi_x+B_int_x,
195  2.0*B_phi_x+B_int_x,
196  BC_int_x,
197  x,
198  MUF2,
199  AlphaS);
200  }
201  if ( flags & (F_EVAL_D_QG_CONT) )
202  {
203  res_qg = Eval_DIP_QG_CONT(s12_1,
204  t11_x,
205  t12_x,
206  B_phi_x+B_int_x,
207  BC_int_x,
208  x,
209  MUF2,
210  AlphaS);
211  }
212  return 1;
213 }
