67 
68  // these have to be adjusted before calls to Eval_??? whenever S changes!!!
69  //HiggsBosons::ResetHiggsPrefactors(s12_1);
70  hm.SetHiggsPrefactors(s12_1,1);
71  // I HAVE CHANGED THIS RECENTLY, CHECK IF SOMETHING IS WRONG
72  // PHIxQCD Born interference terms
73  double B_phi = 0.0;
74  if (flags & F_EVAL_B_PHIxPHI)
75  {
76  B_phi += Eval_B_PHIxPHI(ps,ap,hp);
77  }
78  double B_int = 0.0;
79  if (flags & F_EVAL_B_PHIxQCD)
80  {
81  B_int += Eval_B_2PHIxQCD(ps,ap,hp);
82  }

85  double const& beta_1 = ps.get_beta();

90  double const& S12_1 = 2.0*sp(k1,k2);
91  double const& t11_1 = ps.get_t11();
92  double const& t12_1 = ps.get_t12();

103  AS);
104  res += Eval_DIP_DELTA_PK(s12_1,
105  t11_1,
106  t12_1,
107  B_phi+B_int,
108  2.0*B_phi+B_int,// this is the Born contribution that is multiplied with the K-bar terms
109  BC_int,
110  MUF2,MUR2,
111  AS);
112  }
113  // distribution end-point part
114  double const& x = ps.get_x();
115  if (flags & F_EVAL_D_END && x<Cuts::IDIP_X_CUT)
116  {
117  res += Eval_DIP_DIST_E(s12_1,
118  t11_1,
119  t12_1,
120  B_phi+B_int,
121  2.0*B_phi+B_int,// this is the Born contribution that is multiplied with the K-bar terms
122  BC_int,
123  x,
124  MUF2,
125  AS);
126  }
127 
128  // the IF dipoles give a P+K contribution for PHI^2, although the colour correlations are zero
129  // there is a part in the K operator that is proportional to the uncorrelated Born MEs
130  // there, the factor 1/2 has to becompensated for the PHI^2 part
131  // this factor was introduced because half of the dipoles contructed from the interference terms only are neglected
132 
133  return res;
134 }
