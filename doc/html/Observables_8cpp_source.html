90 // cosine of Collin-Soper angle defined by top quark and proton beam direction in tt z.m.f.
91 double OBS_Theta1(const PS_2* ps_lab, const PS_2* ps_tt)
92 {
93 #ifdef DEBUG
94  if (ps_tt == nullptr)
95  {
96  ERROR_NOEXIT("nullptr received in observable");
97  return 0.0;
98  }
99 #endif
100  // in case of 2->n reactions with n>2, the tt system can have nonzero transverse momentum
101  // we have to compute the vector that dissects the angle between P1 and -P2 (proton momenta)
102  if (ps_tt->whattype()>22)
103  {
104  // this is just the sum of the normalized vectors
105  FV N = ps_tt->P1().norm3() - ps_tt->P2().norm3();
106 
107  if (LEN(N)<1e-10)
108  {
109  WARNING("LEN(N)<1e-10");
110  }
111  PRINT_4VEC(N);
112  PRINT(obs_PHI(N,ps_tt->P1()));
113  PRINT(obs_PHI(N,ps_tt->P2()));
114  FV K = ps_tt->k1()+ps_tt->k2();
115  FV Q = ps_tt->P1()+ps_tt->P2();
116  PRINT_4VEC(K);
117  PRINT_4VEC(Q);
118  return obs_PHI(ps_tt->k1(),N);
119  }
120  return obs_PHI(ps_tt->k1(),ps_tt->P1());
121 }
