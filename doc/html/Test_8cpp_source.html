92  int h,
93  HiggsModel& model,
94  double const& mt_MS,
95  bool NNLO
96  )
97 {
98  using namespace Constants;
99 
100  double const& as = model.AlphaS();
101  double const& mt_pole= model.mt();
102  double const& mb_MS = model.mb();
103  HPtr phi = model.GetBoson(h);
104  double const& mu = model.MUR();//phi->M();
105 
106  PRINT(mu);
107  PRINT(as);
108  PRINT(mt_pole);
109  PRINT(mt_MS);
110  PRINT(mb_MS);
111 
112  double ks = 1.0;
113  double kp = 1.0;
114  if (NNLO)
115  {
116  int nf = 5;
117  // running top mass for the radiative corrections
118  ks = Ks(as,mu,phi->M(),mt_MS,nf);
119  kp = Kp(as,mu,phi->M(),mt_MS,nf);
120  }
121 
122  // top pole mass and bottom MS-bar mass for evaluation of the 1-loop form factors
123  phi->SetFormFactors(phi->M2(),pow(mt_pole,2),pow(mb_MS,2));
124  c_double Fs1 = phi->GetFs()*(-4.0)*phi->M2();
125  c_double Fp1 = phi->GetFp()*8.0*phi->M2();
126 
127  return 1.0/(32.0*Pi)*CF*CA/(phi->M())*pow(as/Pi,2)*(ks*norm(Fs1)+kp*norm(Fp1));
128 }
