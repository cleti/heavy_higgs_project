66 // computes the phi -> gg decay rate inlcuding NNLO QCD corrections in the large mt limit if NNLO = true
67 static double Gamma_hgg(
68  int h,
69  HiggsModel& model,
70  double const& mt_MS,
71  bool NNLO
72  )
73 {
74  using namespace Constants;
75 
76  double const& as = model.AlphaS();
77  double const& mt_pole= model.mt();
78  double const& mb_pole= model.mb();
79  HPtr phi = model.GetBoson(h);
80  double const& mu = phi->M();
81 
82  double ks = 1.0;
83  double kp = 1.0;
84  if (NNLO)
85  {
86  int nf = 5;
87  if (mu>mt_pole) nf = 6;
88  PRINT(nf);
89  // running MS-bar mass for the radiative corrections
90  ks = Ks(as,mu,phi->M(),mt_MS,nf);
91  kp = Kp(as,mu,phi->M(),mt_MS,nf);
92  }
93 
94  // pole masses for evaluation of the 1-loop form factors
95  phi->SetFormFactors(phi->M2(),pow(mt_pole,2),pow(mb_pole,2));
96  c_double Fs1 = phi->GetFs()*(-4.0)*phi->M2();
97  c_double Fp1 = phi->GetFp()*8.0*phi->M2();
98 
99  return 1.0/(32.0*Pi)*CF*CA/(phi->M())*pow(as/Pi,2)*(ks*norm(Fs1)+kp*norm(Fp1));
100 }
